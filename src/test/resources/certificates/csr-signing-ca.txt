# Use openssl req -x509 to generate a self-signed CA cert + private key directly in PEM format.

openssl req -x509 -newkey rsa:2048 -sha256 -days 3650 -nodes \
  -keyout src/test/resources/certificates/csr-signing-ca-key.pem \
  -out src/test/resources/certificates/csr-signing-ca-cert.pem \
  -subj "/CN=SteVe Test CSR Signing CA/O=SteVe Test CA/OU=QA/C=DE" \
  -addext "basicConstraints=critical,CA:TRUE" \
  -addext "keyUsage=critical,keyCertSign,cRLSign,digitalSignature"

# Verify the results with:

openssl x509 -in src/test/resources/certificates/csr-signing-ca-cert.pem -noout -text

# and checked it shows:
# - Basic Constraints: CA:TRUE
# - Key Usage: Certificate Sign, CRL Sign
